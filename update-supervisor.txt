#!/bin/sh

version=$(curl -s http://version.muthur-command.com/stable.json | jq -r '.supervisor // "latest"' || echo "latest")
arch=$(docker image ls | grep -- '-mcio-observer'  | awk '{print $1}' | awk -F/ '{print $3}' | awk -F- '{print $1}')

docker pull "ghcr.io/muthur-command/${arch}-mcio-supervisor:${version}"
docker tag "ghcr.io/muthur-command/${arch}-mcio-supervisor:${version}" "ghcr.io/muthur-command/${arch}-mcio-supervisor:latest"

# mcio-supervisor script will recreate the container automatically.
docker stop mcio_supervisor
